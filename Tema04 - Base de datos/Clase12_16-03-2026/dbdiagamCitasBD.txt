Table especialidades {
   id integer [primary key, increment]
  nombre varchar(150)
}

Table doctores {
  id integer [primary key, increment]
  nombre varchar(150)
  especialidad_id integer
  fecha_registro timestamp
}

Table citas {
  doctor_id integer
  paciente_id integer
  fecha_cita timestamp
  estado char(1)
  fecha_registro timestamp
}

Table pacientes {
  id integer [primary key, increment]
  nombre varchar(150)
  correo varchar(150)
  telefono varchar(150)
}

// Relación: muchos a muchos
Ref: citas.doctor_id > doctores.id
Ref: citas.paciente_id > pacientes.id
Ref: doctores.especialidad_id > especialidades.id
